      SUBROUTINE NORMA(X,XNORM,XMOD)
C
C     GIVEN 3-D VECTOR X, CALCULATES NORMALISED (UNIT) VECTOR XNORM
C     AND MODULUS OF X AS XMOD
C
      IMPLICIT DOUBLE PRECISION (A-H,O-Z)
      DIMENSION X(3),XNORM(3)
C
      CALL VECMOD(X,XMOD)
C
      IF (XMOD.LT.1E-10) THEN
         A=0
      ELSE
         A=1/XMOD
      END IF
C
      DO 10 K=1,3
        XNORM(K)=A*X(K)
10    CONTINUE
C
      END
